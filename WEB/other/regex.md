#### 正则校验值类型

- 日期时间格式校验

  ```js
  // 校验示例,括号内为示例 ([12:13])、([12:13],)、([12:13],[12:13],)
  let reg = /^\[(([1-9])|([0-1][0-9])|([1-2][0-3])):([0-5][0-9])\]$/;
  
  // 校验示例,括号内为示例 ([12:12-13:13])、([12:12-13:13],)、([12:12-13:13],[12:12-13:13],)
  let reg = /(^(\[(([1-9])|([0-1][0-9])|([1-2][0-3])):([0-5][0-9])-(([1-9])|([0-1][0-9])|([1-2][0-3])):([0-5][0-9])\])$)/;
  
  // 校验示例,括号内为示例 ([1212-1213])、([0122-0913],)、([1122-1213],[0522-0713],)
  let reg =  /^\[(([0][1-9])|([1][0-2]))(([0][1-9])|([1-2][0-9])|([3][0-1]))-(([0][1-9])|([1][0-2]))(([0][1-9])|([1-2][0-9])|([3][0-1]))\]$/;
  
  // 校验示例,括号内为示例 ([0-9])、([1011-3023])、([1011-3023],)、([1011-3023],[1011-3023])
  let reg = /^\[([0-9]{1,})-([0-9]{1,})\]$/;
  
  
  // 使用
  let msg = "[12:12],[13:13],"; // 示例
  let result = msg.split(","); // 以英文逗号拆开多个值
  let valid = true; // 校验结果默认为true
  if (result.length) {
    result.forEach((ele) => {
      if (ele && (!reg.test(ele))) {
        valid = false; // 有一次false时，校验结果为false
        console.log(valid); 
      }
    });
  }
  ```
